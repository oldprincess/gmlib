    分组密码Cipher

    一、类型定义

    file: gmlib/types.h

    每个分组密码算法都实现下述函数并定义相应的 CipherInfo 结构变量，用于工作模式的调用

    • 算法上下文(Context)初始化函数
    typedef void (*CipherInit)(uint8_t* key, void* ctx);
    • 算法加密函数
    typedef void (*CipherEncrypt)(uint8_t* out, uint8_t* in, void* ctx);
    • 算法解密函数
    typedef void (*CipherDecrypt)(uint8_t* out, uint8_t* in, void* ctx);
    • 算法信息(Info)结构体
    typedef struct CipherInfo {
    CipherInit init;        ///< 初始化算法
    CipherEncrypt encrypt;  ///< 加密算法
    CipherDecrypt decrypt;  ///< 解密算法
} CipherInfo;

    二、密码算法

    2.1 SM4

    file: gmlib/cipher/sm4.h

    参考资料

    1. GM/T 0002-2012 SM4分组密码算法
    2. 郎欢, 张蕾, 吴文玲. SM4的快速软件实现技术
    • 算法信息(Info)结构体
    #define SM4_KEYLEN 16
#define SM4_BLOCK_SIZE 16

extern const CipherInfo SM4Info;
    • SM4算法上下文(Context)结构
    typedef struct SM4_CTX {
    uint32_t rk[32];  // 32轮轮密钥
} SM4_CTX;
    • 算法上下文(Context)初始化函数
    void sm4_init(uint8_t* key, SM4_CTX* ctx);
    • 算法加密函数
    void sm4_encrypt(uint8_t* out, uint8_t* in, SM4_CTX* ctx);
    • 算法解密函数
    void sm4_decrypt(uint8_t* out, uint8_t* in, SM4_CTX* ctx);

    Example

    static uint8_t sm4_key[SM4_KEYLEN] = {
    0x01, 0x23, 0x45, 0x67, 0x89, 0xab, 0xcd, 0xef,
    0xfe, 0xdc, 0xba, 0x98, 0x76, 0x54, 0x32, 0x10,
};

static uint8_t sm4_pt[SM4_BLOCK_SIZE] = {
    0x00, 0x11, 0x22, 0x33, 0x44, 0x55, 0x66, 0x77,
    0x88, 0x99, 0xaa, 0xbb, 0xcc, 0xdd, 0xee, 0xff,
};

// 128比特密文输出
uint8_t out[SM4_BLOCK_SIZE];

SM4_CTX sm4key;
sm4_init(sm4_key, &sm4key);
sm4_encrypt(out, sm4_pt, &sm4key);
// 09 32 5c 48 53 83 2d cb 93 37 a5 98 4f 67 1b 9a
dump_data(out, SM4_BLOCK_SIZE);

    三、工作模式

    file: gmlib/cipher/mode.h

    参考资料

    1. SP 800-38A Recommendation for Block Cipher Modes of Operation: Methods and Techniques
    2. SP 800-38D Recommendation for Block Cipher Modes of Operation: Galois/Counter Mode (GCM) and GMAC
    3. D. McGrew and J. Viega. The Galois/Counter Mode of Operation (GCM)

    Note

    该版本定义的工作模式仅支持密码分组长度 BLOCK_SIZE 为16字节(128比特)的分组密码算法

    3.1 ECB模式

    • 工作模式上下文(Context)
    typedef struct ECB_CTX {
    uint8_t buffer[BLOCK_SIZE];  // 缓冲区
    int bsize;                   // 当前缓冲区大小
    const CipherInfo* cipher;    // 算法参数
    void* cctx;                  // 算法上下文
} ECB_CTX;
    • 模式上下文(Context)初始化函数
    void ecb_init(uint8_t* key,              ///< [in]    用户密钥
              const CipherInfo* cipher,  ///< [in]    算法
              void* cctx,                ///< [inout] 算法上下文
              ECB_CTX* mctx);
    • 模式加密函数
    void ecb_encrypt_update(uint8_t* out,  ///< [out] 输出
                        int* outl,     ///< [out] 输出长度
                        uint8_t* in,   ///< [in]  输入
                        int inl,       ///< [in]  输入长度
                        ECB_CTX* mctx);

    void ecb_encrypt_final(uint8_t* out, int* outl, ECB_CTX* mctx);

    Note

    执行 init 函数后，多次 update 输入数据，最后 final 处理填充

    • 模式解密函数
    void ecb_decrypt_update(uint8_t* out,  ///< [out] 输出
                        int* outl,     ///< [out] 输出长度
                        uint8_t* in,   ///< [in]  输入
                        int inl,       ///< [in]  输入长度
                        ECB_CTX* mctx);

    int ecb_decrypt_final(uint8_t* out, int* outl, ECB_CTX* mctx);

    Example

    #include <gmlib/cipher/mode.h>
#include <gmlib/cipher/sm4.h>
#include <gmlib/err.h>
#include <gmlib/utils.h>

static uint8_t sm4_key[SM4_KEYLEN] = {
    0x01, 0x23, 0x45, 0x67, 0x89, 0xab, 0xcd, 0xef,
    0xfe, 0xdc, 0xba, 0x98, 0x76, 0x54, 0x32, 0x10,
};

static uint8_t sm4_pt[SM4_BLOCK_SIZE] = {
    0x00, 0x11, 0x22, 0x33, 0x44, 0x55, 0x66, 0x77,
    0x88, 0x99, 0xaa, 0xbb, 0xcc, 0xdd, 0xee, 0xff,
};

static uint8_t out[SM4_BLOCK_SIZE * 2];

int main() {
    SM4_CTX sm4key;
    ECB_CTX ctx;
    uint8_t* outptr = out;
    int outl;
    ecb_init(sm4_key, &SM4Info, &sm4key, &ctx);
    ecb_encrypt_update(outptr, &outl, sm4_pt, sizeof(sm4_pt), &ctx);
    outptr += outl;
    ecb_encrypt_final(outptr, &outl, &ctx);
    outptr += outl;
    // 09 32 5c 48 53 83 2d cb 93 37 a5 98 4f 67 1b 9a
    // 00 2a 8a 4e fa 86 3c ca d0 24 ac 03 00 bb 40 d2
    dump_data(out, (int)(outptr - out));

    return 0;
}

    3.3 GCM模式

    • 工作模式上下文(Context)

    Note

    GCM模式上下文 GCM_CTX 由GHash上下文 GHash_CTX 和GCTR上下文 GCTR_CTX 共同构成。GHashTable 是GHash操作的查找表，在GCM的 init 函数中可选择传入对应指针或是 NULL，传入指针则采用查表优化，否则采用普通实现

    // GHashTable 将占用巨大存储空间
typedef uint64_t GHashTable[256][2];

typedef struct GHash_CTX {
    uint8_t buffer[BLOCK_SIZE];  // 缓冲区
    int bsize;                   // 当前缓冲区大小
    uint8_t X[BLOCK_SIZE];       // GHash当前状态X
    uint8_t H[BLOCK_SIZE];       // 参数H
    GHashTable* ht;              // 查找表指针
} GHash_CTX;

typedef struct GCTR_CTX {
    uint8_t j0[BLOCK_SIZE];      // 初始向量
    uint8_t j[BLOCK_SIZE];       // 初始向量
    uint8_t buffer[BLOCK_SIZE];  // 缓冲区
    int bsize;                   // 当前缓冲区大小
    const CipherInfo* cipher;    // 算法参数
    void* cctx;                  // 算法上下文
} GCTR_CTX;

typedef struct GCM_CTX {
    GHash_CTX hctx;  // GHash上下文
    GCTR_CTX gctx;   // GCTR上下文
    int alen;        // AAD长度
    int clen;        // 密文长度
} GCM_CTX;

    Note

    GCM模式的大致流程为：调用 init 函数初始化，调用 update_aad 函数传入AAD数据，接着 update 和 final 进行加解密

    • 模式上下文(Context)初始化函数
    void gcm_init(uint8_t* key,              ///< [in]    用户密钥
              uint8_t* iv,               ///< [in]    初始向量
              int ivlen,                 ///< [in]    初始向量长度
              GHashTable* ht,            ///< [inout] GHash查找表(可为NULL)
              const CipherInfo* cipher,  ///< [in]    算法
              void* cctx,                ///< [inout] 算法上下文
              GCM_CTX* mctx);
    • GCM输入AAD(additional authenticated data)
    void gcm_update_aad(uint8_t* aad, int alen, GCM_CTX* mctx);
    • 模式加密函数
    void gcm_encrypt_update(uint8_t* out,  ///< [out] 输出
                        int* outl,     ///< [out] 输出长度
                        uint8_t* in,   ///< [in]  输入
                        int inl,       ///< [in]  输入长度
                        GCM_CTX* mctx);

    void gcm_encrypt_final(uint8_t* out,  ///< [out] 输出
                       int* outl,     ///< [out] 输出长度
                       uint8_t* tag,  ///< [out] tag
                       int tlen,      ///< [in]  tag长度
                       GCM_CTX* mctx);
    • 模式解密函数
    void gcm_decrypt_update(uint8_t* out,  ///< [out] 输出
                        int* outl,     ///< [out] 输出长度
                        uint8_t* in,   ///< [in]  输入
                        int inl,       ///< [in]  输入长度
                        GCM_CTX* mctx);

    int gcm_decrypt_final(uint8_t* out,  ///< [out] 输出
                      int* outl,     ///< [out] 输出长度
                      uint8_t* tag,  ///< [in]  tag
                      int tlen,      ///< [in]  tag长度
                      GCM_CTX* mctx);
