uint32_t loadu32_be(uint8_t *src);¶

以大端形式读取32比特数

void storeu32_be(uint8_t *dst, uint32_t n);¶

以大端形式存储32比特数

void storeu64_be(uint8_t *dst, uint64_t n);¶

以大端形式存储64比特数

example¶
uint8_t b[4] = {0x12, 0x34, 0x56, 0x78};
uint32_t n = loadu32_be(b);
printf("n=0x%08x\n", n);  // n=0x12345678
storeu32_be(b, 0x87654321);
// 87 65 43 21
for (int i = 0; i < 4; i++) {
    printf("%02x ", b[i]);
}
puts("");
